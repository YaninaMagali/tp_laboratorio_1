LinkedList* ll_subList(LinkedList* this,int from,int to)
{
    LinkedList* cloneArray = NULL;
    int i;
    void* pElement;


    if(this != NULL && from>=0 && from<to && to<ll_len(this))
    {
        cloneArray = ll_newLinkedList();

        if(cloneArray != NULL)
        {
            for(i=from;i<=to;i++)
            {
                pElement = ll_get(this, i);
                if(pElement != NULL)
                {
                    ll_add(cloneArray, pElement);
                }
            }
        }
    }

    return cloneArray;
}
